// ✅ Function to Validate Amount (Triggers on blur & form submission)
function validateAmount() {
    let amount = $("#amount").val().trim();
    if (amount === "") {
        $("#amount_error").show().text("Amount field cannot be empty.");
        return false;
    }

    if (!/^\d{1,16}(\.\d{1,2})?$/.test(amount)) {
        $("#amount_error").show().text("Amount must be a number up to 16 digits before decimal and 2 after decimal (e.g. 9999999999999999.99). ");
        return false;
    }
    
    let [integerPart, decimalPart] = amount.split(".");
    decimalPart = decimalPart || "00"; // Ensure decimal part exists

    if (integerPart.length === 16) {
        amount = integerPart + "." + decimalPart.padEnd(2, "0"); // Ensure exactly 16 digits + .00
        $("#amount").val(amount);
        $("#amount_error").hide();
    } else if (parseInt(integerPart, 10) >= 50000) {
        amount = integerPart + "." + decimalPart.padEnd(2, "0"); // Ensure min ₹50,000 with .00
        $("#amount").val(amount);
        $("#amount_error").hide();
    } else {
        $("#amount_error").show().text("Amount must be at least ₹50,000 or exactly 16 digits.");
        return false;
    }

    return true;
}
